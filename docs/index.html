    ZetaPush JavaScript SDK

    Install

    From bower

    bower install zetapush-js@2.0.0-beta.7 --save

    <script src="/bower_components/zetapush-js/dist/zetapush.js"></script>

    From npm

    npm install zetapush-js@2.0.0-beta.7 --save

    import { Client } from 'zetapush-js/lib/client'

    From CDN

    <script src="https://static.zpush.io/js/2.0.0-beta.7/zetapush.min.js"></script>

    Usage

    const client = new ZetaPush.Client({
  sandboxId: '<YOUR-SANDBOX-ID>',
  handshakeStrategy() {
    return ZetaPush.AuthentFactory.createWeakHandshake({
      token: null,
      deploymentId: '<YOUR-DEPLOYMENT-ID>'
    })
  }
})
